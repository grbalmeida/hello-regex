package cod3r;

/*
 * Multiline comment
 */

public class Student {
    // Inline comment

    public static void main(String[] args) {
        int age = 20;
        boolean isMaria;

        if(age == 20) {
            isMaria = true;
            System.out.println("Maria");
        } else {
            isMaria = false;
            System.out.println("Mariana");
        }

        for(long counter = 0; counter <= 20; counter++) {
            System.out.println(counter);
        }

        while(false) {
            System.out.println("False");
        }
    }
}
